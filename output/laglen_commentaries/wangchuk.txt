རྒྱལ་བ་ -འི་ སྲས་ ཀྱི་ ལག་ལེན་ སོ་བདུན་ མ -འི་ ཚིག་འགྲེལ་ རྒྱལ་སྲས་ ལམ་ བཟང་ གསལ་བ་ -འི་ སྒྲོན་མེ་ ཞེས་ བྱ་བ་ བཞུགས་ སོ །_། རྟེན་འབྲེལ་ མངོན་ གཟིགས་ སྣང་སྟོང་ སྒྱུ་འཕྲུལ་ གར་ །_། རྟག་ ཁྱབ་ ལྷུན་གྲུབ་ མཛད་པ་ མཐའ་ -ར་ ཕྱིན་ ཀྱང་ །_། ལྔ་ བདོ -འི་ འགྲོ་ ལ་ རྗེས་ཆགས་ ཐུགས་ སྟོབས་ ཀྱིས །_། འགྲོ་ འདྲེན་ བླ་མེད་ ཤཱཀྱ -འི་ དབང་པོ་ -ར་ འདུད་ །_། ལྟ་སྤྱོད་ དགོངས་པ་ ཤིང་རྟ་ རྣམ་ གཉིས་ ལས་ །_། ལེགས་ བརྒྱུད་ རྒྱལ་སྲས་ ལམ་ བཟང་ གསལ་ མཛད་པ་ -འི །_། བཀའ་གདམས་ ལམ་ འབྲས་ བརྒྱུད་པ་ -འི་ བླ་མ་ དང་ །_། ཁྱད་པར་ ཐོགས་མེད་ བཟང་པོ་ -འི་ ཞབས་ ལ་ འདུད་ །_། རྒྱལ་བ་ -འི་ དགོངས་པ་ ཟབ་ རྒྱ་ གནད་བསྡུས་ དོན་ །_། མ་ ནོར་ ཇི་བཞིན་ བསྡུས་པ་ -འི་ ལེགས་བཤད་ འདི་ -ར །_། གོ་ སླ་ -འི་ ངག་ གིས་ ཚིག་དོན་ རྣམ་བཤད་པ་ །_། རྒྱལ་སྲས་ ལམ་ བཟང་ གསལ་བ་ -འི་ སྒྲོན་མེ་ འགོད་ །_། དེ་ ལ་ འདི་ -ར་ དུས་ གསུམ་ རྒྱལ་བ་ རྣམས་ ཀྱི་ བགྲོད་པ་ གཅིག་པ་ -འི་ ལམ་ བཟང་ ། རྒྱལ་སྲས་ བྱང་ཆུབ་ སེམས་དཔའ་ རྣམས་ ཀྱིས་ ཉམས་ སུ་ བླང་ བྱ་ -འི་ ལག་ལེན་ ། གནས་སྐབས་ དང་ མཐར་ཐུག་ གི་ བདེ་བ་ ཀུན་ གྱི་ འབྱུང་ཁུངས་ ། འཁོར་བ་ དང་ མྱ་ངན་ ལས་ འདས་པ་ -འི་ ལེགས་ཚོགས་ ཐམས་ཅད་ ཀྱི་ གཞི་རྩ་ ། རྒྱལ་སྲས་ ཆེན་པོ་ དངུལ་ཆུ་ ཐོགས་མེད་ དཔལ་ བཟང་པོ་ -ས་ མཛད་པ་ -འི ། རྒྱལ་སྲས་ ལག་ལེན་ སོ་བདུན་ མ་ ཞེས་ བྱ་བ་ -འི་ གཞུང་ བཟང་པོ་ ཡིད་བཞིན་ གྱི་ ནོར་བུ་ ལྟ་བུ་ འདི་ཉིད་ འཆད་པ་ -ར་ བྱེད་པ་ ལ་ གསུམ་ ། ཐོག་མ་ -ར་ དགེ་བ་ བསྟན་བཅོས་ རྩོམ་པ་ -འི་ རྒྱུ་ བསྟན་པ་ དང་ ། བར་ དུ་ དགེ་བ་ བརྩམ་ བྱ་ བསྟན་བཅོས་ ལུས་ ཀྱི་ རང་བཞིན་ དངོས་ ། ཐ་མ་ -ར་ དགེ་བ་ བསྟན་བཅོས་ བརྩམ་པ་ མཇུག་ གྲུབ་པ་ -འི་ བྱ་བ་ བཤད་པ་ དང་ གསུམ་ ལས་ ། དང་པོ་ ལ -འང་ གསུམ་ ། མཚན་དོན་ དང་ །

དེ་བཞིན་ དུ་ འབྲོམ་སྟོན་པ་ -ས་ ། སངས་རྒྱས་ འདྲ་བ་ -འི་ སྟོན་པ་ མེད་ །_། ཆོས་ དང་ ། འདྲ་བ་ -འི་ སྐྱོབ་པ་ མེད་ །_། དགེ་འདུན་ འདྲ་བ་ -འི་ ཞིང་ས་ མེད་ །_། དེ་ཕྱིར་ བདག་ གིས་ སྐྱབས་ སུ་ བགྱི་ །_། ཞེས་ སོགས་ ཀྱི་ ཕྱིར་ ན་ ། དཀོན་མཆོག་ གསུམ་ གཏན་ གྱི་ སྐྱབས་ སུ་ བསྟེན་ ན་ གནས་སྐབས་ ལྷ་ དང་ མི་ -འི་ བདེ་སྐྱིད་ ནས་ བཟུང་ སྟེ་ མཐར་ཐུག་ སངས་རྒྱས་ ཀྱི་ གོ་འཕང་ སོགས་ ཐོབ་པ་ -ས་ ན་ སྒོ་གསུམ་ དད་གུས་ དང་ བཅས་ ཏེ་ སྐྱབས་ སུ་ འགྲོ་བ་ -ར་ བྱ་ དགོས་ ཏེ་ ། རྒྱལ་སྲས་ ཉིད་ ཀྱིས་ ། གང་ ལ་ སྐྱབས་ ན་ གནས་སྐབས་ མཐར་ཐུག་ གི་ ། བདེ་བ་ ཐམས་ཅད་ ངེས་པར་ སྩོལ་ ནུས་པ་ -འི་ །_། དཀོན་མཆོག་ གསུམ་ ལ་ བློ་ ལྟོས་ ལེགས་ གཏད་ ནས་ །_། གུས་པ་ -འི་ ཡིད་ ཀྱིས་ སྐྱབས་འགྲོ་ དྲག་ ཏུ་ བྱ་ །_། ཞེས་ གསུངས་ སོ་ །_། དེ་ལྟར་ སྐྱབས་ སུ་ སོང་བ་ -འི་ ཕན་ཡོན་ ཡང་ ཕར་ཕྱིན་ བསྡུས་པ་ ལས་ ། སྐྱབས་ སོང་ བསོད་ནམས་ གཟུགས་ མཆིས་ ན་ །_། སྟོང་གསུམ་ འདི་ ཡང་ སྣོད་ དུ་ ཆུང་ །_། རྒྱ་མཚོ་ ཆེན་པོ་ ཆུ་ ཡི་ གཏེར་ །_། ཕོར་ གྱིས་ གཞལ་བ་ -ར་ ག་ལ་ ནུས་ །_། ཞེས་ སོ་ །_། དེ་ -འང་ དཀོན་མཆོག་ ཅེས་པ་ ནི་ དཀོན་པ་ ལས་ ཀྱང་ ཤིན་ཏུ་ དཀོན་པ་ ལ་ གོ་ དགོས་པ་ སྟེ་ ། སྤྱིར་ འཇིག་རྟེན་ ན་ དཀོན་པ་ -འི་ དངོས་པོ་ མང་ སྟེ་ ། པདྨ་ བཀའ་ཐང་ ལས་ ། མེ་ཏོག་ མང་ སྟེ་ ཨུ་དུམ་ཝ་ར་ དཀོན ནིརྒྱ་ མཚོ་ མང་ སྟེ་ འོ་མ་ ཅན་ ནི་ དཀོན་ ༔ འདབ་ཆགས་ མང་ སྟེ་ མཁའ་ལྡིང་ ཁྱུང་ཆེན་ དཀོན་ ༔ སེང་གེ་ མང་ སྟེ་ གངས་ ཀྱི་ སེངྒེ་ དཀོན་ ཞེས་ སོགས་ གསུངས་པ་ ལྟར་ དུ་ དཀོན་པ་ -འི་ དངོས་པོ་ མང་ ཡང་ ། དེ་དག་ གིས་ ཀྱང་ གཞལ་ དུ་ མེད་པ་ -ར་ ཆེས་ དཀོན་པ་ -ས་ ན་ དཀོན་མཆོག་ ཅེས་ གསུངས་ སོ་ །_། དེ་ -འང་ ཇི་སྐད་དུ་ ། འབྱུང་བ་ དཀོན་ ཕྱིར་ དྲི་མེད་ ཕྱིར་ །_།
