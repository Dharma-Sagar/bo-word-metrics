ད་རེས་ ཀྱི་ ལུས་ འདི་ -འི་ སྟེང་ ནས་ ཕ་ ༈ མར་ གྱུར་པ་ -འི་ སེམས་ཅན་ ཐམས་ཅད་ ཀྱི་ ཆེད་ དུ་ བདག་ གིས་ ཚེ་ དང་ སྲོག་ ལ་ ཐུག་ ཀྱང་ སྐྲག་ འཇིགས་ མི་བྱེད་པ་ -ར་ དགའ་སྤྲོ་ -འི་ ངང་ ནས་ གཞན་ཕན་ འབའ་ཞིག་ དོན་ དུ་ གཉེར་ ན་ ནི་ དེ་ -འི་ འབྲས་བུ་ མཐར་ཐུག་ ཆོས་ གཟུགས་ ཀྱི་ སྐུ་ གཉིས་ མངོན་དུ་ གྱུར་ ནས་ ཕ་མ་ སེམས་ཅན་ རྣམས་ ལ་ རྟག་ ཁྱབ་ ལྷུན་གྲུབ་ ཀྱི་ མཛད་པ་ དུས་འཁོར་ བ་མ་ སྟོངས་ བར་ དུ་ བྱེད་ ནུས་པ་ སྟེ་ ། དཔེར་ན་ མཁའ་དབྱིངས་ ཟླ་བ་ གཅིག་ གི་ བཞིན་རས་ ཀྱིས་ འཛམ་གླིང་ ཆུ་སྣོད་ ཀུན་ ལ་ ཁྱབ་པ་ ལྟ་བུ་ -འི་ འགྲོ་དོན་ བྱེད་ ནུས་པ་ ཡིན་ ནོ་ །_། རྒྱུ་མཚན་ དེ་ -འི་ ཕྱིར་ ན་ བདག་ གི་ བདེ་བ་ ཐམས་ཅད་ སེམས་ཅན་ གཞན་ ལ་ སྦྱིན་ ཞིང་ །_། གཞན་ མཁའ་ཁྱབ་ ཀྱི་ སེམས་ཅན་ རྣམས་ ཀྱི་ སྡུག་བསྔལ་ མཐའ་དག་ རང་ སྟེང་ དུ་ བླངས་ ནས་ བདེ་སྡུག་ གཉིས་ཀ་ ཕན་ཚུན་ ཡང་དག་པ་ -ར་ བརྗེ་བ་ ནི་ རྒྱལ་སྲས་ བྱང་ཆུབ་སེམས་དཔའ་ རྣམས་ ཀྱིས་ བོདག་ གཞན་ བརྗེ་བ་ བསྒོམ་ ཚུལ་ གྱི་ ཉམས་ སུ་ བླང་ བྱ་ -འི་ ལག་ལེན་ ཡིན་ ནོ་ །_། འདི་ -ར་ ཅུང་ འཕྲོ་ -ས་ ན་ བདག་ གཞན་ བརྗེ་བ་ ཞེས་པ་ -འི་ དོན་ དེ་ མདོར་བསྡུས་ ན་ ། གཙོ་བོ་ ནི་ རང་ གཅེས་པ་ -ར་ འཛིན་པ་ དང་ ། གཞན་ ཡལ་བ་ -ར་ དོར་བ་ -འི་ བློ་ གཉིས་ བརྗེས་ ནས་ ། གཞན་ ལ་ རང་ ལྟར་ གཅེས་པ་ -ར་ འཛིན་ ཞིང་ ། རང་ ལ་ གཞན་ ལྟར་ ཡལ་བ་ -ར་ འདོར་བ་ -འི་ བློ་ བསྐྱེད་ དེ་ བདག་ ལ་ གཞན་ གྱི་ སྡིག་ སྡུག་ ཐམས་ཅད་ སྨིན་ ཞིང་ ། བདག་ གི་ བདེ་ དགེ་མ་ ལུས་པ་ གཞན་ ལ་ སྦྱིན་ དགོས་ ཏེ་ ། རིན་ཆེན་ ཕྲེང་བ་ ལས་ ། བདག་ ལ་ དེ་དག་ སྡིག་ སྨིན་ ཞིང་ །_། བདག་ དགེ་མ་ ལུས་ དེ་ -ར་ སྨིན་ ཤོག་ ། ཅེས་ དང་ ། སྤྱོད་འཇུག་ ལས་ ། བདག་ཉིད་ གཞན་ ལ་ བཏང་བ་ དང་ །_།
