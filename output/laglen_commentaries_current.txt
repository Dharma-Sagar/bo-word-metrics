དེ་བཞིན་ དུ་ ཉོན་མོངས་པ་ ཞེ་སྡང་ དང་ གཏི་མུག་ ཕྲག་དོག་ ང་རྒྱལ་ སོགས་ གང་ ཡིན་ ཀྱང་ ཁོ་རང་ གི་ དབང་ དུ་ བཞག་ ནས་ ཅི་ཙམ་ ལོངས་ སྤྱད་ ཀྱང་ དེ་ -ས་ མི་ཆོག་པ་ -ར་ མ་ཟད་ ། སླར་ཡང་ སྲེད་པ་ འཕེལ་བ་ -ར་ འགྱུར་བ་ -ས་ ན་ སྲིད་པ་ -འི་ ཕུན་ཚོགས་ ལ་ ཆགས་པ་ ནི་ འདི་ དང་ ཕྱི་མ་ -འི་ ཉེས་དམིགས་ ཤིན་ཏུ་ མང་པོ་ -འི་ འབྱུང་གནས་ ཡིན་ ནོ་ །_། དེ་ལྟར་ ཡང་ མདོ་ རྒྱ་ཆེར་རོལ་པ་ ལས་ ། འདོད་པ་ -འི་ ཡོན་ཏན་ དག་ ལ་ དགའ་བ་ ནི་ །_། ལན་ཚྭ་ -འི་ ཆུ་ འཐུང་ བཞིན་ དུ་ ངོམས་ མི་འགྱུར་ །_། ཞེས་ གསུངས་ སོ་ །_། དེ་ལྟར་ འདོད་པ་ -འི་ ལོངས་སྤྱོད་ ལ་ ཆགས་ ན་ དགྲ་ ལ་སོགས་པ་ -ས་ འཇིགས་པ་ མང་བ་ དང་ ། གནོད་པ་ ཤིན་ཏུ་ མང་བ་ -ས་ སྡུག་བསྔལ་ ཐམས་ཅད་ ཀྱི་ འབྱུང་ཁུངས་ ཡིན་ ཏེ་ ། དཔེར་ན་ རལ་གྲི་ -འི་ ཁ་ -འི་ སྦྲང་རྩི་ ལྟ་བུ་ དང་ ། དུག་ སྡོང་ གི་ ལོ་མ་ དང་ ། ངན་ སྐྱུག་ གིས་ སྦགས་པ་ -འི་ སྣོད་ ལྟ་བུ་ ཡིན་པ་ -ས་ ན་ ། (འཕགས་ ལམ་/ དོན་ དུ་ གཉེར་བ་ རྣམས་ ཀྱིས་ རིང་ དུ་ སྤོང་ དགོས་ ཏེ་ ། རྒྱ་ཆེར་རོལ་པ་ ལས་ ། འདོད་པ་ -འི་ ཡོན་ཏན་ མྱ་ངན་ གནོད་པ་ མང་ །_། འཇིགས་ བཅས་ རྟག་ ཏུ་ དགྲ་ བྱེད་ ཉོན་མོངས་ བཅས་ །_། རལ་གྲི་ -འི་ སོ་ འདྲ་ དུག་ སྡོང་ ལོ་མར་ མཚུངས་ །_། ངན་ སྐྱུག་ སྣོད་ ལྟར་ འཕགས་པ་ -འི་ སྐྱེ་བོ་ -ས་ སྤངས་ །_། ཞེས་ སོ་ །_། དེས་ན་ འདོད་པ་ -འི་ ལོངས་སྤྱོད་ ཇི་ལྟར་ ཡོད་ ཀྱང་ ། དེ་ ལ་ ཞེན་ ན་ ཆོག་ཤེས་ ཡོད་ མི་སྲིད་ དེ་ ། དཔེར་ན་ གསེར་དངུལ་ གྱི་ ཆ་ -ར་ བབས་ ཀྱང་ ཆགས་ ལྡན་ གྱི་ གང་ཟག་ ནི་ མི་ ངོམས་པ་ -འི་ སྟེང་ དུ་ ད་དུང་ སྲེད་པ་ ལྷག་པ་ -ར་ འཕེལ་ ཞིང་ སྡུག་བསྔལ་ བར་ འགྱུར་ རོ་ །_། དེ་ལྟར་ འདོད་པ་ -འི་ ལོངས་སྤྱོད་ ནི་ མནོག་ ཆུང་ ཞིང་ ཉེས་པ་ མང་བ་ །
